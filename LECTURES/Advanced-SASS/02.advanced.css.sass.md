## Hidden Declarations

- Sometimes you only want a property declaration to show up some of the time.

```<sass>
$rounded-corners: false;

.button {
    border: 1px solid black;
    border-radius: if($rounded-corners, 5px, null);
}
```

## default Values

```<sass>
    /// !default value is used when not overriden
    _variables.scss;
    $default-color: #fff !default;

    _usage.scss
    .heading {
        color: $default-color; /// #fff is used
    }

    _override.scss
    $default-color: red;
    .headeing {
        color: $default-color; /// red is used
    }
```

## Shadowing

- Use keyword !global to shadow variables

```<sass>
    $variable: first global value;
    .header {
        $variable: second global value !global;
        value: $variable; /// output: second global value
    }
    .main {
        value: $variable; /// output: second global value
    }
```

## Flow Control Scope

```<sass>
$dark-them: false;
$primeery-color: red !default;

@if $dark-them {
    $primery-color: darken($primery-color, 50%);
}
```

## Advance Variable Functions

```<sass>
@use 'sass:map';
$theme-colors: {
    'success': blue,
    'info': red,
    'warning': yellow
};
.button {
    background-color: map.get($theme-color, 'warning');
}
```

## SASS Data Types

```<sass>
/// numbers
$font-size: (
    'normal': 100,
    'decimals': 0.1,
    'percentage': 100%,
    'pxValue': 16.5px,
    'rem': 1rem
);

/// colors
$color: {
    named: red, /// do not use
    specific-name: midnightblue,
    hexa: #fff,
    hex_alpha: #b373939aa,
    rgb: rgb(207, 1, 0),
    rgba: rgba(207, 1, 0, 0.5)
}

/// SASS List
- List contain a sequence of other values. In SASS, elements in lists can be separated by commas (hel, Arial, sans-serif).
$list-1: 2rem, 3rem, 4rem;
$background-color: green, blue, red;

@each $color in $background-color {
    .list-item-#{$color} { /// .list-item-green
        background-color: $color;
    }
}

/// Maps holds key/values
$theme-colors: (
    'success': blue,
    'info': red,
    'warning': yellow
);

$color: (
    named: red, /// do not use
    specific-name: midnightblue,
    hexa: #fff,
    hex_alpha: #b373939aa,
    rgb: rgb(207, 1, 0),
    rgba: rgba(207, 1, 0, 0.5)
);

p {
    background-color: map-get($theme-colors, 'success');
}

.Map-list {
    .list-item {
        font-size: 1.5rem;
        &:first-child {
            color: map-get($theme-color, success);
        }
        &:nth-child(2) {
            color: map-get($theme-color, info);
        }
        &:nth-child(3) {
            color: map-get($theme-color, warning);
        }
    }
}

/// Merge maps
$bg_map: map-merge($theme-color, $color);
p {
    color: map-get($bg_map, success);
    background-color: map-get($bg_map, named);
}

/// Booleans
@mixin avatar($size, $circle: true) {
    $width: $size;
    height: $size;
    @if $circle {
        border-radius: $size / 2;
    }
}

.square {
    background-color: red;
    @include avatar(100px); /// default value for circle is true
    @include avatar(100px, false); /// default value for circle is false
}

/// SASS Operators
/// Equality Operators
@debug 1px == 1px;  /// true
@debug 21px == 1px; /// false
@debug hsl(34, 35%, 92.1%) == #f0ece41; /// true
@debug 'Helvetica' == Helvetica; /// true
@debug (5px 7px 10px) == (5px 7px 10px); /// true

// Relational Operators
@debug 100 > 50;
@debug 100 < 50;
@debug 100 >= 50;
@debug 100 <= 50;

// Numeric Operators
@debug 10s + 15s; /// 25s
@debug 1in - 10px; // 0.8958in
@debug 5px * 3px; /// 15px
@debug (12px/4px); /// 3
@debug 1in % 9px; /// 0.0625in

// String Operators
@debug "Helvetica" + "Neue"; // Helvetica Neue
@debug sans- + serif; /// sans-serif
@debug #{10px+5px}; /// 15px/5px

// Boolean Operators
@debug not true; /// false
@debug not false; /// true
@debug true and true; /// true
@debug true and false; /// false
@debug true or false; /// true

/// Flow Control Rules Overview - Use in Mixins
@if and @else
@mixin Title($title) {
    @if expression {
        @include title;
    } else if $direction == down {

        @error "Unknown direction #{$direction}";
    }
}

h3 {
    &:first-child {
        @include Title($title: false);
    }
}
```
